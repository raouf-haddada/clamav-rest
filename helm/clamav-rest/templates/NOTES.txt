ClamAV REST API has been deployed.

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "clamav-rest.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "clamav-rest.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.httpPort }}
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "clamav-rest.fullname" . }} {{ .Values.service.httpPort }}:{{ .Values.service.httpPort }}
{{- end }}

2. Test the connection:
  curl -i http://localhost:{{ .Values.service.httpPort }}/

3. Scan a file:
  curl -F 'file=@/path/to/file' http://localhost:{{ .Values.service.httpPort }}/v2/scan

{{- if .Values.tls.enabled }}

TLS is enabled. HTTPS is available on port {{ .Values.service.httpsPort }}.
{{- else }}

TLS is disabled. Only HTTP is available. Enable it with --set tls.enabled=true.
{{- end }}
